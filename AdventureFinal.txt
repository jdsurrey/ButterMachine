@echo off
setlocal EnableExtensions EnableDelayedExpansion
title BRIMBLE SPARKSTEEL: SLASHER OF THE FROST KING - Chapter 1

goto :START

rem -------------------- Helpers --------------------
:SCENE
cls
echo ======================================================================
echo  %~1
echo ======================================================================
echo.
exit /b

:PAUSE
echo.
pause
exit /b

:CHOICE
set "cvar=%~1"
set "cprompt=%~2"
set "callowed=%~3"
set "cans="
:CHOICE_LOOP
set /p "cans=%cprompt% "
if not defined cans goto :CHOICE_LOOP
set "cans=%cans:~0,1%"
for /f "delims=%callowed%" %%A in ("%cans%") do (
  echo.
  echo Invalid choice. The dungeon is judging you.
  echo.
  goto :CHOICE_LOOP
)
set "%cvar%=%cans%"
exit /b

rem NOTE: All deaths must pass a message; if none is passed we generate a unique one.
rem This avoids "ECHO is off." and prevents repeated stock lines.
:DEATH
set "DMSG=%~1"
if not defined DMSG call :RANDOM_DEATHLINE
call :SCENE "YOUR FINAL MOMENTS (IN GLORIOUS, OVERPRODUCED HD)"
echo !DMSG!
echo.
echo Press any key to continue...
pause >nul
call :SCENE "YOU DIED (THE AUDIENCE IS DELIGHTED AND ALSO A LITTLE SICK)"
echo The synth sting hits so hard it files for custody.
echo The camera zooms in past your pores and into your regrets.
echo The audience cheers, booing in harmony like a haunted choir.
echo.
echo The SHOW-NARRATOR delivers the recap voice-over with a velvet grin:
echo "TRAGIC. CAMPY. PROFITABLE."
echo "WE WILL REPLAY THAT IN SLOW MOTION DURING THE ADS."
echo "YOUR FINAL EXPRESSION HAS BEEN LICENSED TO A LUNCHBOX."
echo.
echo Press any key to begin again from Chapter 1...
pause >nul
endlocal
start "" "%~f0"
exit

:DIE
call :DEATH "%~1"
exit /b

rem Unique fallback death lines (only used if someone forgets to pass a message).
:RANDOM_DEATHLINE
set /a _dl=!RANDOM! %% 12
if "!_dl!"=="0" set "DMSG=You perish heroically in the same way a man perishes after saying 'watch this.' The dungeon records it in 4K and immediately adds a laugh track."
if "!_dl!"=="1" set "DMSG=You discover the ancient truth: the fastest way out is usually down. Gravity accepts your offering with gratitude and no refunds."
if "!_dl!"=="2" set "DMSG=Your courage spikes briefly, then is replaced by consequences. The consequences win by unanimous decision."
if "!_dl!"=="3" set "DMSG=You die like a true professional: on camera, in costume, and with absolutely zero control over the soundtrack."
if "!_dl!"=="4" set "DMSG=You make one small, reasonable choice. The dungeon responds by escalating to murder, because it is emotionally immature."
if "!_dl!"=="5" set "DMSG=You become a cautionary tale told to future contestants. Children will chant your name at summer camps for centuries."
if "!_dl!"=="6" set "DMSG=You attempt competence. The universe flags it as suspicious activity and removes you accordingly."
if "!_dl!"=="7" set "DMSG=You go out not with a bang but with a contractual synth sting. Somewhere, an intern wins a bet."
if "!_dl!"=="8" set "DMSG=You die as you lived: confused, damp, and being observed by unseen critics with snacks."
if "!_dl!"=="9" set "DMSG=You perish because you trusted something labeled 'safe.' The label is promoted to executive producer."
if "!_dl!"=="10" set "DMSG=You are defeated by the oldest villain of all: the plot. It has home-field advantage and a better agent."
if "!_dl!"=="11" set "DMSG=You die dramatically, but the lighting is so flattering the audience briefly forgets to be sad."
exit /b
